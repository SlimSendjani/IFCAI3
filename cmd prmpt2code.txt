Get-ChildItem -Recurse -File -Include "*.js","*.html","*.css","*.svg" | Where-Object { $_.FullName -notlike "*\node_modules\*" -and $_.Name -notlike "package-lock.json" -and $_.Name -notlike ".gitattributes" -and $_.Name -notlike "Feuille de route générale" } | ForEach-Object { "//***********************************"; "// Path: $($_.FullName.Substring((Get-Location).Path.Length))"; "//***********************************"; Get-Content $_.FullName; "" } | Out-File -FilePath "output.txt" -Encoding UTF8
